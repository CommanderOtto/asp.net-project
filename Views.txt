//View for Advisor Status 
GO
IF OBJECT_ID('VW_KBS_ADVISOR_STATUS') IS NOT NULL
  EXEC('DROP VIEW VW_KBS_ADVISOR_STATUS')
GO
CREATE VIEW VW_KBS_ADVISOR_STATUS
AS
SELECT ASID, AStaDesc
FROM KBS_ADVISOR_STATUS
WHERE AStaActive = 1;


//View for Appointment Status
GO
IF OBJECT_ID('VW_KBS_APPOINTMENT_STATUS') IS NOT NULL
  EXEC('DROP VIEW VW_KBS_APPOINTMENT_STATUS')
GO
CREATE VIEW VW_KBS_APPOINTMENT_STATUS
AS
SELECT STID, StaDesc
FROM KBS_APPOINTMENT_STATUS
WHERE StaActive = 1;


//View for Majors
GO
IF OBJECT_ID('VW_KBS_MAJORS') IS NOT NULL
  EXEC('DROP VIEW VW_KBS_MAJORS')
GO
CREATE VIEW VW_KBS_MAJORS
AS
SELECT MID, MajName
FROM KBS_MAJORS
WHERE MajActive = 1;


//View for Reasons
GO
IF OBJECT_ID('VW_KBS_REASONS') IS NOT NULL
  EXEC('DROP VIEW VW_KBS_REASONS')
GO
CREATE VIEW VW_KBS_REASONS
AS
SELECT RID, ReaDesc
FROM KBS_REASONS
WHERE ReaActive = 1;
